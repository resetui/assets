project.ts 
import { ResetUI } from "resetui"

const resetui = new ResetUI()

const res = await resetui.create({
  url: "https://example.com",
  prompt: "Bold monochrome SaaS redesign",
  source: "https://example.com",
    export: "html",
    versions: true,
  }),
  
})

for await (const chunk of res) {
  // yields sections, edits, and export metadata
  process.stdout.write(chunk.text)
}
